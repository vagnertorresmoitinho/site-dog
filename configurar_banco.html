🐕 Configurar Banco de Dados - DogTop

⚠️ IMPORTANTE: O banco não tem as tabelas necessárias. Execute o SQL abaixo no SQL Editor do Supabase para criar as tabelas.

Passo 1: Acesse o Supabase

1. Acesse https://supabase.com/dashboard
2. Selecione o projeto anjuawcodnktbjsznpeb
3. No menu lateral, clique em SQL Editor
4. Copie o código abaixo e cole no editor
5. Clique em Run para executar

Passo 2: SQL para criar tabelas

-- ============================================
-- SCHEMA DOGTOP - MODO ONLINE
-- ============================================

-- PRODUTOS
create table produtos (
    id uuid primary key default gen_random_uuid(),
    codigo text,
    nome text not null,
    categoria text,
    descricao text,
    preco_base numeric(10, 2) not null default 0,
    preco_promocional numeric(10, 2),
    custo_producao numeric(10, 2) default 0,
    estoque integer default 0,
    tempo_preparo_minutos integer default 15,
    imagem text,
    ingredientes text[] default '{}',
    tags text[] default '{}',
    ativo boolean not null default true,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);

-- ADICIONAIS
create table adicionais (
    id uuid primary key default gen_random_uuid(),
    nome text not null,
    descricao text,
    preco_venda numeric(10, 2) not null default 0,
    custo_estimado numeric(10, 2) default 0,
    imagem text,
    ativo boolean not null default true,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);

-- CLIENTES
create table clientes (
    id uuid primary key default gen_random_uuid(),
    nome text not null,
    cpf_cnpj text,
    email text,
    ddi_celular text default '+55',
    telefone_celular text,
    ddi_telefone text default '+55',
    telefone_fixo text,
    endereco text,
    numero text,
    bairro text,
    cep text,
    complemento text,
    ponto_referencia text,
    status text default 'Ativo',
    saldo numeric(10, 2) default 0,
    observacoes text,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);

-- ENTREGADORES
create table entregadores (
    id uuid primary key default gen_random_uuid(),
    nome text not null,
    telefone text,
    veiculo text,
    placa text,
    ativo boolean not null default true,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);

-- PEDIDOS
create table pedidos (
    id uuid primary key default gen_random_uuid(),
    cliente_id uuid references clientes(id),
    nome_cliente text,
    telefone text,
    tipo_entrega text default 'Presencial',
    endereco_entrega text,
    bairro text,
    cep text,
    ponto_referencia text,
    forma_pagamento text,
    troco_para numeric(10, 2),
    troco_devolver numeric(10, 2),
    subtotal_original numeric(10, 2) default 0,
    desconto numeric(10, 2) default 0,
    subtotal numeric(10, 2) default 0,
    taxa_entrega numeric(10, 2) default 0,
    total_pedido numeric(10, 2) default 0,
    status text default 'Recebido',
    observacao text,
    origem text default 'pdv',
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);

-- ITENS DO PEDIDO
create table itens_pedido (
    id uuid primary key default gen_random_uuid(),
    pedido_id uuid not null references pedidos(id) on delete cascade,
    produto_id uuid references produtos(id),
    adicional_id uuid references adicionais(id),
    parent_item_id uuid references itens_pedido(id) on delete cascade,
    tipo_item text not null,
    nome_item text not null,
    quantidade integer not null default 1,
    preco_unitario numeric(10, 2) not null default 0,
    custo_unitario numeric(10, 2) default 0,
    observacao text,
    created_at timestamptz default now()
);

-- ENTREGAS
create table entregas (
    id uuid primary key default gen_random_uuid(),
    pedido_id uuid not null references pedidos(id) on delete cascade,
    entregador_id uuid references entregadores(id),
    status text default 'Aguardando',
    saiu_em timestamptz,
    entregue_em timestamptz,
    confirmado_cliente_em timestamptz,
    observacao text,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);

-- CONFIGURAÇÕES
create table configuracoes (
    id uuid primary key default gen_random_uuid(),
    chave text unique not null,
    valor jsonb,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);

-- ÁREAS DE ENTREGA
create table areas_entrega (
    id uuid primary key default gen_random_uuid(),
    cidade text not null,
    bairro text not null,
    taxa_entrega numeric(10, 2) not null default 0,
    ativo boolean default true,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);

-- AUDIT LOGS
create table audit_logs (
    id uuid primary key default gen_random_uuid(),
    acao text not null,
    entidade text not null,
    entidade_id text,
    dados_old jsonb,
    dados_new jsonb,
    usuario text,
    ip text,
    created_at timestamptz default now()
);

-- ÍNDICES
create index idx_produtos_categoria on produtos(categoria);
create index idx_produtos_ativo on produtos(ativo);
create index idx_produtos_nome on produtos(nome);
create index idx_clientes_telefone on clientes(telefone_celular);
create index idx_clientes_status on clientes(status);
create index idx_pedidos_status on pedidos(status);
create index idx_pedidos_cliente_id on pedidos(cliente_id);
create index idx_pedidos_created_at on pedidos(created_at desc);
create index idx_itens_pedido_pedido_id on itens_pedido(pedido_id);
create index idx_entregas_pedido_id on entregas(pedido_id);
create index idx_entregas_status on entregas(status);
